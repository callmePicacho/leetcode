## 思路

基本链表操作，返回值时通过"当前节点"，插入和删除都需要找到待操作节点的"前一节点"

可以通过设置 Size 来快速判断 index 和链表个数的关系

还可以通过初始化时设置虚拟头节点，减少头插头插的特判

## AC 代码

```go
type Node struct {
    Val int
    Next *Node
}

type MyLinkedList struct {
    Head *Node
    Size int
}

func Constructor() MyLinkedList {
    // 设置虚拟头结点，头插头删不必特判
    return MyLinkedList{Head:&Node{}}
}

func (this *MyLinkedList) Get(index int) int {
    if index < 0 || this.Size <= index {
        return -1
    }
    cur := this.Head.Next // 真正的头
    for i := 0; i < index; i++ {
        cur = cur.Next
    }
    return cur.Val
}

func (this *MyLinkedList) AddAtHead(val int)  {
    node := &Node{Val: val, Next: this.Head.Next}
    this.Head.Next = node
    this.Size++
}

func (this *MyLinkedList) AddAtTail(val int)  {
    pre := this.Head
    for pre.Next != nil {
        pre = pre.Next
    }
    node := &Node{Val: val, Next: pre.Next}
    pre.Next = node
    this.Size++
}

func (this *MyLinkedList) AddAtIndex(index int, val int)  {
    if index <= 0 {  // 头插
        this.AddAtHead(val)
    } else if index >= this.Size { // 尾插
        this.AddAtTail(val)
    } else {
        pre := this.Head
        for i := 0 ; i < index; i++ {
            pre = pre.Next
        }
        node := &Node{Val: val, Next: pre.Next}
        pre.Next = node
        this.Size++
    }
}

func (this *MyLinkedList) DeleteAtIndex(index int)  {
    if index < 0 || index >= this.Size {
        return 
    }
    pre := this.Head
    for i := 0 ; i < index; i++ {
        pre = pre.Next
    }
    pre.Next = pre.Next.Next
    this.Size--
}
```
